begin:
	movement move1 moveLoop1 reverseMove1 reverseLoopLabel1 finishLabel1
	quickFire
	movement move2 moveLoop2 reverseMove2 reverseLoopLabel1 finishLabel2
	accurateAddFire
	jmp begin
	
macro movement moveLabel loopLabel reverseLabel reverseLoopLabel finishLabel
	%moveLabel%:
		rnd 8
		swp
		seta 50
		move 2
	%loopLabel%:
		jmpc %reverseLabel%
		;jmpm %loopLabel%
		;jmp %finishLabel%
		wait 3
		jmp %finishLabel%
	%reverseLabel%:
		seta -30
		setb 48
		move -1
		%reverseLoopLabel%:
			;jmpm %reverseLoopLabel%
			wait 1
		jmp %moveLabel%
	%finishLabel%:
/macro

macro quickFire
	scan
	lock
	fire
	fire
/macro

macro accurateAddFire
	lock
	fire
/macro
